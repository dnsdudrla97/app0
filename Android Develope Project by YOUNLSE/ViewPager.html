뷰페이저

  • 손가락으로 좌우 스크롤하여 넘겨볼 수 있는 기능을 제공한다.
  • 화면 전체를 뷰 페이지로 채운다면 좌우 스크롤을 통해 화면을 넘겨볼 수 있게 된다.
  • 화면 일부분만 차지하고 있어도 그부분에서만 좌우 스크롤이 동작한다.
  • 뷰페이저는 그 안에 프래그먼트를 넣을 수 있고 좌우 스크롤도 프래그먼트를 전환하게 된다.
  • 뷰페이저는 내부에서 어댑터라는 것과 상호박용하게 되어 있는데 이것은 뷰페이저가 여러 개의 아이템 중에 하나를 보여주는 방식으로 동작하기 때문이다.

타이틀스트립(TitleStrip)

  • 지금 보고 있는 화면이 전체 프래그먼트들 중에서 몇 번째 것인지 알기는 어렵다.
  • 그래서 위나 아래쪽에 전체 아이템의 개수와 현재 보고 있는 아이템이 어떤 것인지를 보여줄 필요가 있다.
  • 이런 역활을 하는 것인 타이틀스트립, 탭스트립을 사용할 수 있다.

MainActivity.java

package com.younsle.samplepager;

import androidx.annotation.NonNull;
import androidx.appcompat.app.AppCompatActivity;
import androidx.fragment.app.Fragment;
import androidx.fragment.app.FragmentManager;
import androidx.fragment.app.FragmentStatePagerAdapter;
import androidx.viewpager.widget.ViewPager;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;

import java.util.ArrayList;

public class MainActivity extends AppCompatActivity {
    ViewPager pager;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        Button button = findViewById(R.id.button);
        button.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                pager.setCurrentItem(1);
            }
        });

        pager = findViewById(R.id.pager);
        //ViewPager 객체를 findViewById() 메서드로 찾아 참조한 후 클래스 안에 선언한 변수에 할당했다.
        pager.setOffscreenPageLimit(3);
        //setOffscreenPageLimit() 메서드로 미리 로딩해 놓을 아이템의 개수를 세 개로 늘렸다.
        //뷰페이저는 어댑터가 가지고 있는 아이템 중에서 몇 개를 미리 로딩해 두었다가 좌우 스크롤할 때 빠르게 보여줄수 있다.

        MyPagerAdapter adapter = new MyPagerAdapter(getSupportFragmentManager());

        Fragment1 fragment1 = new Fragment1();
        adapter.addItem(fragment1);

        Fragment2 fragment2 = new Fragment2();
        adapter.addItem(fragment2);

        Fragment3 fragment3 = new Fragment3();
        adapter.addItem(fragment3);

        pager.setAdapter(adapter);
        //VeiwPager 객체에 어댑처 객체를 설정하고 있다. -> setAdapter() 메서드를 호출하면 어댑터 객체가 설정되고 이때부터 뷰페이지 어댑터에 있는 프래그먼트들을 화면에 보여줄 수 있게 된다.
        //!만약 사용자가 손가락으로 화면을 전환하지 않고 코드에서 전환시키고 싶다면 뷰페이저 객체의 setCurrentItem() 메서드를 사용하면된다.
    }
    //MyPagerAdapter 클래스는 내부 클래스로 만들어 졌으며 FragmentStatePagerAdapter를 상속한다.
    //어댑터는 뷰페이지에 보여줄 각 프래그먼트를 관리하는 역활을 하며, 뷰페이저에 설정하면 서로 상호작용하면서 화면에 표시해준다.
    class MyPagerAdapter extends FragmentStatePagerAdapter {
        ArrayList<Fragment> items = new ArrayList<Fragment>();
        //MyPagerAdapter 안에는 프래그먼트들을 담아둘 ArrayList 객체를 만들었으며 그 안에는 프래그먼트 객체를 넣어둘 것이다.
        public MyPagerAdapter(FragmentManager fm){
            super(fm);
        }

        public CharSequence getPageTitle(int position) {
            return "페이지" + position;
        }
        //PaperTitleStrip

        public void addItem(Fragment item) {
            items.add(item);
            //addItem() -> ArrayList 객체 안에 프래그먼트를 추가할 수 있는 메서드
        }

        @NonNull
        @Override
        public Fragment getItem(int position) {
            return items.get(position);
            //getCount() -> 프래그먼트의 개수를 확인할 수 있다.
            //뷰페이저는 어댑터와 상호작용하면서 getCount() 메서드로 몇 개의 프래그먼트가 들어 있는지 확인한다.
            //그런 다음 화면의 상태에 따라 해당하는 프래그먼트를 꺼내와 보여주게 된다. -> 이 어댑터를 사용할 수 있도록 다음 코드를 onCreate() 메서드에 추가
        }

        @Override
        public int getCount() {
            return items.size();
        }
    }
}
